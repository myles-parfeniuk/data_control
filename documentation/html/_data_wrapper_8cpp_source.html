41 {
42 //initialize new follower
43 Follower follower = {
44 .cb = cb, //assign callback
45 .f_id = next_follower_id, //assign id
46 .paused = false, //follower unpaused by default
47 };
48
49 next_follower_id++; //generate id for next follower
50
51 if(logging_en)
52 ESP_LOGI(TAG, "%s -> Follow: %d", name, follower.f_id);
53
54 if(!execute_immediately)
55 {
56 lock_follower_list();
57 follower_list.push_back(follower); //push new follower to list
58 unlock_follower_list();
59 }
60 else

62 lock_immediate_follower_list();
63 immediate_follower_list.push_back(follower); //push new immediate follower to list
64 unlock_immediate_follower_list();
65 }
66
67
68 return follower.f_id; //return the id of the new follower
